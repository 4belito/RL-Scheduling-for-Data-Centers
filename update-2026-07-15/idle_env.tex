% ---- State space: Timed (Idle-Releases) environment --------
\begin{frame}{Timed Environment: State Space}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
      \begin{block}{Context (global)}
        \footnotesize
        \begin{itemize}\setlength\itemsep{2pt}
          \item \texttt{free nodes}
        \end{itemize}
      \end{block}
      \begin{block}{Per running job \;(\textcolor{SoftCoral}{$V_r=4$} soonest-ending)}
        \footnotesize
        \begin{itemize}\setlength\itemsep{2pt}
          \item \texttt{nodes held}
          \item \texttt{remaining time to release}
        \end{itemize}
      \end{block}
    \column{0.48\textwidth}
      \begin{block}{Per queued job \;(\textcolor{SoftCoral}{$V_q=10$} visible)}
        \footnotesize
        \begin{itemize}\setlength\itemsep{2pt}
          \item \texttt{nodes required}
          \item \texttt{wall time}
        \end{itemize}
      \end{block}
      \vspace{0.5em}
  \end{columns}
\vspace{1.6em}
    {\small\textcolor{DeepBlue}{{\textbf{Reward:} schedulable-idle penalty:}\\ $$r=-\dfrac{\min(D_t,\;n_{\text{free}})}{N}$$}}
  {\footnotesize\textcolor{SoftCoral}{\textbf{Note:} the non-timed env has \emph{no running block} (only \texttt{free nodes} + \texttt{nodes required}); with no running queue, cross-attention is not used.}}
\end{frame}
